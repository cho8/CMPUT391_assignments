PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX db: <http://dbpedia.org/>
PREFIX dbc: <http://dbpedia.org/resource/Category:>
PREFIX dct: <http://purl.org/dc/terms/>
PREFIX schema: <http://schema.org/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
PREFIX dbo: <http://dbpedia.org/ontology/>
SELECT ?city sum(?population)/sum(?num_hosp) as ?pop_ratio WHERE {
SELECT ?city ?population (count(?hospital)) as ?num_hosp WHERE {
  ?city dct:subject dbc:Cities_in_Alberta .		# Cities in Alberta
  ?city rdf:type schema:City .
  ?hospitals skos:broader dbc:Hospitals_in_Canada_by_province .
  ?hospitalcat skos:broader ?hospitals .
  ?hospital dct:subject ?hospitalcat .
  ?hospital dbo:region ?city .
  ?hospital dbo:bedCount ?population
}
} GROUP BY ?city
ORDER BY DESC(?pop_ratio)
